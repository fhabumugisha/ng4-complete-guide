    description: string;
    imagePath: string;
    ingredients: Ingredient[];
    constructor (id: number, name: string, desc: string, imagePath: string, ingredients: Ingredient[]) {
        this.id = id;
        this.name = name;
        this.description = desc;
        this.imagePath =  imagePath;
        this.ingredients = ingredients;
    }
}
